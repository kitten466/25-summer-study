const myMap = new Map();

myMap.set("이름", "철수");
myMap.set("나이", 30);
myMap.set(123, "숫자 키");

console.log(myMap.get("이름"));
console.log(myMap.get(123));

console.log(myMap.has("나이"));

for (const [key, value] of myMap) {
console.log(key, ":", value);
}

myMap.delete("이름");
console.log(myMap);
